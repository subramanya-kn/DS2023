class Turbine {

	float rotationSpeed;
	int size;
	String generatorType;

	public void setTurbine(float rotationSpeed, int size, String generatorType) {
		this.rotationSpeed = rotationSpeed;
		this.size = size;
		this.generatorType = generatorType;
	}

	void printTurbine() {
		System.out.println("The roatation speed is :" + rotationSpeed);
		System.out.println("Size is :" + size);
		System.out.println("Geneartor type is :" + generatorType);
	}

}
